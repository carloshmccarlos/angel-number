name VARCHAR(255),

birthday DATE,

focus_area VARCHAR(50),

name_number INT,

life_path_number INT,

personal_vibration INT,

language VARCHAR(10),

device_type VARCHAR(20),

utm_source VARCHAR(100),

created_at TIMESTAMP DEFAULT NOW()

CREATE TABLE email_leads (

id BIGSERIAL PRIMARY KEY,
